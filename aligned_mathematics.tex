\begin{align}
    (x + h) ^ 2 - x ^ 2 & = x ^ 2 + 2 x h + h ^ 2 - x ^ 2 && \text{by distributivity} \\
                        & = 2xh + h ^ 2 && \text{by subtraction}\nonumber\\
                        & = h(2x+h) && \text{by factorisation}\label{my_favourite_step}
\end{align}
